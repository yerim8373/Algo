	public static long solution(int n, int[] works) {
		long answer = 0;
		PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());

		for (int i = 0; i < works.length; i++) {
			pq.offer(works[i]);
		}

		while(n > 0) {
			int work = pq.poll();
			if(work == 0) break;
			work--;
			pq.offer(work);
			n--;
		}

		for(int q : pq) {
			answer += Math.pow(q, 2);
		}

		return answer;
	}
}
